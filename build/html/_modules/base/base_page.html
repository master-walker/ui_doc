[文档]    @typeassert(locator=tuple, condition=str)
    def find_element(self, locator, sleep_time=wait_time, condition=None):
        """
        查找页面元素

        参数类型需要匹配：locators：tuple, condition：str

        :Args:
         - locator - 定位元素 格式示例：(By.ID, 'element_id', "元素信息") 或 (By.ID, 'element_id')，
            元素信息可选，主要用于打印调试日志以及后期维护
         - sleep_time - 可选参数
            显式等待时间，默认读取配置时间
         - condition - 可选参数
            expected_conditions模块提供的等待条件

        :Usage:
         _locator = (By.ID, 'element_id', "元素信息")
         self.find_element(self._locator)
         self.find_element(self._locator, sleep_time=5, condition='element_to_be_clickable')

        :Returns:
         返回Webelement对象
        """
        self.logger.debug(f'定位的元素：{locator}')
        start = time.time()
        try:
            wait = WebDriverWait(self.driver, sleep_time)
            if condition and hasattr(EC, condition):
                element = wait.until(getattr(EC, condition)(locator[:2]))
            else:
                element = wait.until(EC.presence_of_element_located(locator[:2]))
            self.logger.info("获取元素 {0} {1}, Spend {2} seconds".format(locator[-1], success, time.time() - start))
            return element
        except (NoSuchElementException, TimeoutException) as e:
            self.logger.error("获取元素 {0} {1}, Spend {2} seconds".format(locator[-1], fail, time.time() - start))
            self.logger.error(traceback.print_exc())



[文档]    def find_elements(self, locators, sleep_time=wait_time):
        """
        根据locators获取一个元素或多个元素

        主要为操作多个元素或单个元素的方法提供支持

        :Args:
         - locators: 定位元素, 参数类型：元祖或数组
            格式示例：(By.ID, 'element_id', "元素信息") 或 [(By.ID, 'element_id', "元素信息"),(By.ID, 'element_id', "元素信息")]，
            元素信息参数可选，主要用于打印调试日志以及后期维护
         - sleep_time - 可选参数
            显式等待时间，默认读取配置时间

        :Usage:
         _locator1 = (By.ID, 'element_id', "元素信息")
         _locator2 = (By.ID, 'element_id', "元素信息")
         _locators = [_locator1, _locator2]
         self.find_elements(self._locator1)
         self.find_elements(self.locators, sleep_time=5)

        :Returns:
         locators参数为单个元素，返回Webelement对象
         locators参数是多个元素组成的数组，返回多个Webelement对象组成的数组
        """
        self.logger.debug(f'定位的元素：{locators}')
        if locators and isinstance(locators, tuple):
            element = self.find_element(locators, sleep_time)
            self.logger.debug("获取到的元素：{0}".format(element))
            return element
        elif locators and isinstance(locators, list):
            if len(locators) == 1:
                return self.find_element(locators[0], sleep_time)
            else:
                elements = []
                for locator in locators:
                    elements.append(self.find_element(locator, sleep_time))
                self.logger.debug("获取到的元素列表：{0}".format(elements))
                return elements
        else:
            raise InvalidArgumentException("参数类型错误，请传入元祖或数组")



[文档]    @typeassert(locator=tuple)
    def find_group_elements(self, locator, sleep_time=wait_time, num=None):
        """
        获取一组相同类型元素

        :Args:
         - locator - 定位元素 格式示例：(By.ID, 'element_id', "元素信息") 或 (By.ID, 'element_id')，
            元素信息可选，主要用于打印调试日志以及后期维护
         - sleep_time - 可选参数
            显式等待时间，默认读取配置时间
         - num - 可选参数
            获取元素的数量，默认为None，返回所有元素

        :Usage:
         _locator = (By.ID, 'element_id', "元素信息")
         self.find_group_elements(self._locator)
         self.find_group_elements(self._locator, sleep_time=5, num=2)

        :Returns:
         返回一组Webelement对象组成的数组
        """
        start = time.time()
        try:
            group_elements = WebDriverWait(self.driver, sleep_time).until(
                EC.presence_of_all_elements_located(locator[:2]))
            # 获取指定个数的元素
            if group_elements and isinstance(num, int):
                elements = group_elements[:num]
                self.logger.info("获取元素 {0} {1} Spend {2} seconds".format(locator[-1], success, time.time() - start))
                self.logger.debug("已获取元素：{0}".format(elements))
                return elements
            return group_elements
        except (TimeoutException, NoSuchElementException):
            self.logger.error("获取元素 {0} {1} Spend {2} seconds".format(locator[-1], fail, time.time() - start))
            self.logger.error(traceback.print_exc())

[文档]    def send_keys(self, locators, data):
        """
        输入框输入操作

         :Args:
         - locators - 定位元素, 参数类型：元祖或数组
            格式示例：(By.ID, 'element_id', "元素信息") 或 [(By.ID, 'element_id', "元素信息"),(By.ID, 'element_id', "元素信息")]，
         - data - 输入数据
            单个输入，传入单个数据即可
            多个输入，传入多个数据即可，类型一般为元祖或数组

        :Usage:
         _locator = (By.ID, 'element_id', "元素信息")
         _locator1 = (By.ID, 'element_id', "元素信息")
         _locator2 = (By.ID, 'element_id', "元素信息")
         _locators = [_locator1, _locator2]
         test_data = [123, 345]
         self.send_keys(_locator, 123)
         self.send_keys(_locators, test_data)

        """
        start = time.time()
        try:
            elements = self.find_elements(locators)
            if isinstance(elements, list) and isinstance(data, (tuple, list)):
                if len(elements) != len(data):
                    raise InvalidArgumentException(f'输入框和输入数据数量不一致, 元素数目：{len(elements)}，输入数据列表数目：{len(data)}')
                for index, element in enumerate(elements):
                    element.clear()
                    element.send_keys(data[index])
                    self.logger.info("{0} send_keys {1}, Spend {2} seconds".format(locators[index][-1], data[index], time.time() - start))
            elif isinstance(elements, WebElement):
                elements.clear()
                elements.send_keys(data)
                self.logger.info("{0} send_keys {1}, Spend {2} seconds".format(locators[-1], data, time.time() - start))
            else:
                raise NoSuchElementException(f'{locators} 未获取到元素')
        except (ElementNotInteractableException, ElementNotVisibleException) as e:
            self.logger.error(traceback.print_exc())



[文档]    def click(self, locator, sleep_time=wait_time):
        """
        点击操作

        :Args:
         - locator - 定位元素 格式示例：(By.ID, 'element_id', "元素信息") 或 (By.ID, 'element_id')，
            元素信息可选，主要用于打印调试日志以及后期维护
         - sleep_time - 可选参数
            显式等待时间，默认读取配置时间

        :Usage:
         _locator = (By.ID, 'element_id', "元素信息")
         self.click(self._locator)
         self.click(self._locator, sleep_time=5)

        """
        start = time.time()
        element = self.find_element(locator, sleep_time=sleep_time, condition='element_to_be_clickable')
        try:
            if isinstance(element, WebElement):
                element.click()
                self.logger.info("点击  {0}  Spend {1} seconds".format(locator[-1], time.time() - start))
                self.sleep(1)
            else:
                self.logger.error(f'{locator}，未获取到元素')
        except UnexpectedAlertPresentException as e:
            self.logger.error(traceback.print_exc())

    def submit(self, locators, data, sleep_time=1):
        """
        输入数据并点击按钮，提交表单

        适用场景：
        多个输入框输入数据后，需要点击提交

        :Args:
         - locators - 定位元素 由至少一个输入框元素和一个按钮元素组成的数组
            格式示例：[(By.ID, 'element_id', "输入框元素"), (By.ID, 'element_id', '提交按钮')]
         - data - 输入数据
            单个输入，则传入一个测试数据
            多个输入，则传入多个测试数据组成的元祖或数组
         - sleep_time - 可选参数
            输入完成后的等待时间，默认1s

        :Usage:
         _locator1 = (By.ID, 'element_id', "输入框1")
         _locator2 = (By.ID, 'element_id', "输入框2")
         _locator = (By.ID, 'element_id', "提交按钮")
         _locators = [_locator1, _locator2, _locator]
         test_data = [123, 345]
         self.submit(_locators, test_data)

        """
        # 输入数据
        self.send_keys(locators[:-1], data)
        self.sleep(sleep_time)
        # 点击提交
        self.click(locators[-1])



[文档]    def find_elements_text(self, locators, sleep_time=wait_time):
        """
        获取一个或多个元素的文本值

        :Args:
         - locator - 定位元素 格式示例：(By.ID, 'element_id', "元素信息") 或 (By.ID, 'element_id')，
            元素信息可选，主要用于打印调试日志以及后期维护
         - sleep_time - 可选参数
            显式等待时间，默认读取配置时间

        :Usage:
         _locator = (By.ID, 'element_id', "元素信息")
         self.find_elements_text(self._locator)
         _locator1 = (By.ID, 'element_id', "元素信息")
         _locator2 = (By.ID, 'element_id', "元素信息")
         _locators = [_locator1, _locator2]
         self.find_elements_text(_locators, sleep_time=5)

        :Returns:
         传入单个元素，返回单个文本
         传入多个元素，返回多个文本组成的数组
        """
        try:
            elements = self.find_elements(locators, sleep_time=sleep_time)
            if elements and isinstance(elements, list):
                text = [element.text for element in elements]
                self.logger.info(f'获取元素的文本： {text}')
            if isinstance(elements, WebElement):
                text = elements.text
                self.logger.info(f'获取元素的文本： {text}')
                return text
        except AttributeError:
            self.logger.error(traceback.print_exc())
