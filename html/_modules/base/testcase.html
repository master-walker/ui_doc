[文档]    @classmethod
    def setup_class(cls):
        """
        环境初始化，整个测试文件执行过程只执行一次
        """
        cls.env = common.get_env_var('env')
        cls.logger.info('   ############################### env: {}; {} start ###############################'.format(cls.env, cls.__name__))
        # 获取驱动
        cls.driver = common.get_driver()
        # 根据配置的运行平台，执行不同的初始化操作
        platform = common.get_env_var('platform')
        if platform in ('web', 'grid'):
            # 默认实例化CommonPage
            cls.comm_page = WebCommonPage(cls.driver)
            # web测试，打开浏览器，窗口最大化
            cls.url = common.get_env_url()
            cls.comm_page.open_and_maximize(cls.url)
        elif platform == 'ios':
            cls.comm_page = IosCommonPage(cls.driver)
        elif platform == 'android':
            cls.comm_page = AndroidCommonPage(cls.driver)
        else:
            raise ValueError('platform可选值：web、android、ios、grid')
